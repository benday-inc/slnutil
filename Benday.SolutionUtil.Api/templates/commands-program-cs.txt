using System.Diagnostics;
using System.Reflection;
using System.Text;
using Benday.CommandsFramework;
using %%API_PROJECT_NAMESPACE%%;

namespace %%PROJECT_NAMESPACE%%;

class Program
{
    static void Main(string[] args)
    {
        CommandsApp
            .Create<SampleAsyncCommand>(args)
            .WithAppSettings()
            .WithConfigFile("appsettings.development.json", optional: true)
            .WithConfigFile("appsettings.unversioned.json", optional: true)
            .WithAppInfo(
                "Sample Tool using Commands Framework", 
                "https://www.benday.com")
            .WithVersionFromAssembly()
            .UsesConfiguration(true)
            .ConfigureOptions(options =>
            {
                // Additional options configuration if needed
                options.StrictArgumentValidation = true;
                options.ConfigurationFolderName = "SampleToolConfig";
            })
            .ConfigureServices(services =>
            {
                // Register your services for dependency injection
                // services.AddSingleton<IGreetingService, GreetingService>();
            })
            .Run();
    }
}
